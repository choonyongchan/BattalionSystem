# Unified Supabase Project Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Consolidate all five company Supabase projects into a single hercules project, with each company's data in prefixed tables (`Archer_NominalRoll`, `Stallion_Exceptions`, etc.) and per-company RLS enforced by email.

**Architecture:** The hercules Supabase project hosts 24 tables (6 company prefixes × 4 table types). A single Supabase JS client is used for all companies. Row-level security gates each table by `auth.email() = '{company}@40sar.internal'`. A `tbl(company, tableName)` helper in `lib/supabase.ts` builds the prefixed table name for component queries.

**Tech Stack:** Next.js 14, Supabase JS v2, TypeScript, Vitest, Playwright, bun

## Global Constraints

- Table names follow `{Company}_{TableName}` — first letter of company capitalised, e.g. `Archer_NominalRoll`
- Companies with prefixed tables: `Archer`, `Braves`, `Cougar`, `Stallion`, `Hercules`, `Test`
- RLS email format: `{lowercase company}@40sar.internal` (e.g. `archer@40sar.internal`)
- Single env var set: `NEXT_PUBLIC_SUPABASE_URL` and `NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY` (no per-company vars)
- Integration tests use `company="hercules"` and auth as `hercules@40sar.internal`; `Test_*` tables exist in schema but are reserved for future use
- Run commands with `bun`, not `npm`/`npx`

```sql
-- Unified schema for all companies in one Supabase project.
-- Each company gets 4 prefixed tables: {Company}_NominalRoll, _Exceptions, _Duty, _Configuration.
-- Run scripts/sync-schema.ts to apply to the hercules project.
-- Safe to re-run (IF NOT EXISTS, ON CONFLICT DO NOTHING).

GRANT USAGE ON SCHEMA public TO anon, authenticated;

DO $$
DECLARE
  companies text[] := ARRAY['Archer','Braves','Cougar','Stallion','Hercules','Test'];
  c text;
BEGIN
  FOREACH c IN ARRAY companies LOOP

    EXECUTE format('
      CREATE TABLE IF NOT EXISTS public."%s_NominalRoll" (
        rank    text NOT NULL,
        name    text NOT NULL,
        platoon text NOT NULL CHECK (platoon = ANY (ARRAY[''HQ'',''1'',''2'',''3'',''4''])),
        PRIMARY KEY (name)
      )', c);

    EXECUTE format('
      CREATE TABLE IF NOT EXISTS public."%s_Exceptions" (
        id      bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
        name    text NOT NULL,
        scope   text NOT NULL CHECK (scope = ANY (ARRAY[''Att C'',''Status'',''Off/Leave'',''Guard Duty'',''Report Sick'',''MA'',''Others''])),
        reason  text NOT NULL,
        start   date NOT NULL,
        "end"   date NOT NULL
      )', c);

    EXECUTE format('
      CREATE TABLE IF NOT EXISTS public."%s_Duty" (
        duty_type text NOT NULL CHECK (duty_type = ANY (ARRAY[''CDO'',''CDS'',''COS'',''PDS1'',''PDS2'',''PDS3'',''PDS4''])),
        date      date NOT NULL,
        name      text NOT NULL,
        PRIMARY KEY (duty_type, date)
      )', c);

    EXECUTE format('
      CREATE TABLE IF NOT EXISTS public."%s_Configuration" (
        parade_type text NOT NULL CHECK (parade_type = ANY (ARRAY[''First Parade'',''Last Parade''])),
        time        time NOT NULL,
        PRIMARY KEY (parade_type)
      )', c);

    EXECUTE format('GRANT SELECT, INSERT, UPDATE, DELETE ON public."%s_NominalRoll"   TO authenticated', c);
    EXECUTE format('GRANT SELECT, INSERT, UPDATE, DELETE ON public."%s_Exceptions"    TO authenticated', c);
    EXECUTE format('GRANT SELECT, INSERT, UPDATE, DELETE ON public."%s_Duty"          TO authenticated', c);
    EXECUTE format('GRANT SELECT, INSERT, UPDATE, DELETE ON public."%s_Configuration" TO authenticated', c);

    EXECUTE format('ALTER TABLE public."%s_NominalRoll"   ENABLE ROW LEVEL SECURITY', c);
    EXECUTE format('ALTER TABLE public."%s_Exceptions"    ENABLE ROW LEVEL SECURITY', c);
    EXECUTE format('ALTER TABLE public."%s_Duty"          ENABLE ROW LEVEL SECURITY', c);
    EXECUTE format('ALTER TABLE public."%s_Configuration" ENABLE ROW LEVEL SECURITY', c);

    EXECUTE format('DROP POLICY IF EXISTS "authenticated can manage %s_NominalRoll"   ON public."%s_NominalRoll"',   c, c);
    EXECUTE format('DROP POLICY IF EXISTS "authenticated can manage %s_Exceptions"    ON public."%s_Exceptions"',    c, c);
    EXECUTE format('DROP POLICY IF EXISTS "authenticated can manage %s_Duty"          ON public."%s_Duty"',          c, c);
    EXECUTE format('DROP POLICY IF EXISTS "authenticated can manage %s_Configuration" ON public."%s_Configuration"', c, c);

    EXECUTE format(
      'CREATE POLICY "authenticated can manage %s_NominalRoll" ON public."%s_NominalRoll"
       FOR ALL TO authenticated
       USING (auth.email() = lower(''%s'') || ''@40sar.internal'')
       WITH CHECK (auth.email() = lower(''%s'') || ''@40sar.internal'')',
      c, c, c, c);

    EXECUTE format(
      'CREATE POLICY "authenticated can manage %s_Exceptions" ON public."%s_Exceptions"
       FOR ALL TO authenticated
       USING (auth.email() = lower(''%s'') || ''@40sar.internal'')
       WITH CHECK (auth.email() = lower(''%s'') || ''@40sar.internal'')',
      c, c, c, c);

    EXECUTE format(
      'CREATE POLICY "authenticated can manage %s_Duty" ON public."%s_Duty"
       FOR ALL TO authenticated
       USING (auth.email() = lower(''%s'') || ''@40sar.internal'')
       WITH CHECK (auth.email() = lower(''%s'') || ''@40sar.internal'')',
      c, c, c, c);

    EXECUTE format(
      'CREATE POLICY "authenticated can manage %s_Configuration" ON public."%s_Configuration"
       FOR ALL TO authenticated
       USING (auth.email() = lower(''%s'') || ''@40sar.internal'')
       WITH CHECK (auth.email() = lower(''%s'') || ''@40sar.internal'')',
      c, c, c, c);

    EXECUTE format('
      INSERT INTO public."%s_Configuration" (parade_type, time) VALUES
        (''First Parade'', ''09:30:00''),
        (''Last Parade'',  ''17:30:00'')
      ON CONFLICT (parade_type) DO NOTHING', c);

  END LOOP;
END;
$$;

GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO authenticated;
```

```typescript
  Row:           { id: number; name: string; scope: string; reason: string; start: string; end: string }
  Insert:        { id?: number; name: string; scope: string; reason: string; start: string; end: string }
  Update:        { id?: number; name?: string; scope?: string; reason?: string; start?: string; end?: string }
  Relationships: []

  Insert:        { duty_type: string; date: string; name: string }
  Update:        { duty_type?: string; date?: string; name?: string }
  Relationships: []

  Row:           { parade_type: string; time: string }
  Insert:        { parade_type: string; time: string }
  Update:        { parade_type?: string; time?: string }
  Relationships: []
}

type CompanyTables =
  { [C in Company as `${Capitalize<C>}_NominalRoll`]:   NominalRollTable } &
  { [C in Company as `${Capitalize<C>}_Exceptions`]:    ExceptionsTable  } &
  { [C in Company as `${Capitalize<C>}_Duty`]:          DutyTable        } &
  { [C in Company as `${Capitalize<C>}_Configuration`]: ConfigTable      }

  Test_NominalRoll:   NominalRollTable
  Test_Exceptions:    ExceptionsTable
  Test_Duty:          DutyTable
  Test_Configuration: ConfigTable
}

type Database = {
  public: {
    Tables:    CompanyTables & TestTables
    Views:     Record<string, never>
    Functions: Record<string, never>
  }
}

// ponytail: company arg kept so call sites don't change; all companies share one project now
const _client = createClient<Database>(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY!,

  return _client
}

export const tbl = (company: Company, table: string) =>

export interface Exception {

  start: string
  end: string
}

  date: string
```

```typescript
/**
 * Applies supabase/schema.sql to the unified hercules Supabase project.
 * Usage: bun run scripts/sync-schema.ts
 *
 * Requires SUPABASE_ACCESS_TOKEN and NEXT_PUBLIC_SUPABASE_URL in .env.local.
 */

import { readFileSync } from 'fs'
import { join, dirname } from 'path'
import { fileURLToPath } from 'url'

const ACCESS_TOKEN = process.env.SUPABASE_ACCESS_TOKEN
if (!ACCESS_TOKEN) {
  console.error('Error: SUPABASE_ACCESS_TOKEN is not set.')
  process.exit(1)
}

const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL
if (!supabaseUrl) {
  console.error('Error: NEXT_PUBLIC_SUPABASE_URL is not set.')
  process.exit(1)
}

const ref = new URL(supabaseUrl).hostname.split('.')[0]
const schemaPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'supabase', 'schema.sql')
const sql = readFileSync(schemaPath, 'utf-8')

process.stdout.write(`Syncing schema to project ${ref}... `)

const res = await fetch(`https://api.supabase.com/v1/projects/${ref}/database/query`, {
  method: 'POST',
  headers: {
    Authorization: `Bearer ${ACCESS_TOKEN}`,
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({ query: sql }),
})

if (!res.ok) {
  const body = await res.text()
  console.log('FAILED.')
  console.error(`HTTP ${res.status}: ${body}`)
  process.exit(1)
}

console.log('done.')
```
